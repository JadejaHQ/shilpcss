## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		table: {
			border: {
				property: "border-collapse: <v><i>;",
				values: {
					collapse: "collapse",
					separate: "separate",
				},
			},
		},
	},
};
```
