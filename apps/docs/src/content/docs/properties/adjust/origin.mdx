## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		adjust: {
			origin: {
				property: "transform-origin: <n><v><i>;",
				resolve: "spacing",
				themeKey: "spacingPixels",
				values: {
					// transform-origin: x y;
					center: "center",
					right: "right",
					left: "left",
					top: {
						DEFAULT: "top",
						left: "left top",
						right: "right top",
					},
					bottom: {
						DEFAULT: "bottom",
						left: "left bottom",
						right: "right bottom",
					},
				},
			},
		},
	},
};

export default shilpConfig;
```
