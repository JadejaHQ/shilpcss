const shilpConfig = {

						1.5: "1.5px",
						2: "2px",

				},

				color: {
					property: "stroke: <v><i>;",
					resolve: "color",
					themeKey: "colors",
					variant: true,
					values: {},
				},

				opacity: {
					property: "stroke-opacity: <v><i>;",
					themeKey: "fractions",
					values: {},
				},

				scale: {
					property: "vector-effect: <v><i>;",
					values: {
						DEFAULT: "none",
						none: "non-scaling-stroke",
					},
				},
			},
		},
	},
};

export default shilpConfig;
