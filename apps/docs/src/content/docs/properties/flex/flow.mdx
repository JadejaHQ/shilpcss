## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		flex: {
			flow: {
				property: "flex-direction: <v><i>;",
				values: {
					row: {
						DEFAULT: "row",
						reverse: "row-reverse",
					},
					col: {
						DEFAULT: "column",
						reverse: "column-reverse",
					},
				},
			},
		},
	},
};

export default shilpConfig;
```
