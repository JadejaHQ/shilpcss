## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		layout: {
			hide: {
				property: "visibility: <v><i>;",
				values: {
					DEFAULT: "hidden",
				},
			},
		},
	},
};

export default shilpConfig;
```
