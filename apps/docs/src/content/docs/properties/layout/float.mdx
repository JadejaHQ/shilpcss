## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		layout: {
			float: {
				DEFAULT: {
					property: "float: <v><i>;",
					values: {
						left: "left",
						right: "right",
						none: "none",
					},
				},

				clear: {
					property: "clear: <v><i>;",
					values: {
						left: "left",
						right: "right",
						none: "none",
						both: "both",
					},
				},
			},
		},
	},
};

export default shilpConfig;
```
