## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		filter: {
			preset: {
				property: "filter: <v><i>;",
				values: {
					// filter: [blur] [brightness] [contrast] [drop-shadow] [grayscale] [hue-rotate] [invert] [opacity] [saturate] [sepia];
				},
			},
		},
	},
};

export default shilpConfig;
```
