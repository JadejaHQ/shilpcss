## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		filter: {
			shadow: {
				DEFAULT: {
					property: "filter: drop-shadow(<v>)<i>;",
					values: {
						none: "0 0 #0000",
						xs: "var(--d-sdw-xs)",
						sm: "var(--d-sdw-sm)",
						DEFAULT: "var(--d-sdw-sm)",
						md: "var(--d-sdw-md)",
						lg: "var(--d-sdw-lg)",
						xl: "var(--d-sdw-xl)",
						"2xl": "var(--d-sdw-2xl)",
					},
				},

				color: {
					property: "--d-sdw-clr: <v><i>;",
					resolve: "color",
					themeKey: "colors",
					variant: true,
					values: {},
				},
			},
		},
	},
};

export default shilpConfig;
```
