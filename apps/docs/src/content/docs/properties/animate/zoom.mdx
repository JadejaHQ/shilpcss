## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		animate: {
			zoom: {
				in: {
					property: "--entr-scl: <n><v><i>;",
					themeKey: "fractions",
					values: {
						DEFAULT: 1,
					},
				},

				out: {
					property: "--ext-scl: <n><v><i>;",
					themeKey: "fractions",
					values: {
						DEFAULT: 1,
					},
				},
			},
		},
	},
};

export default shilpConfig;
```
