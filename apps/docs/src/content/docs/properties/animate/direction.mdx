## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		animate: {
			direction: {
				property: "animation-direction: <v><i>;",
				values: {
					normal: "normal", // from -> to
					reverse: "reverse", // to -> from
					boomerang: "alternate-reverse", // from -> to -> from
					strikeback: "alternate", // to -> from -> to
				},
			},
		},
	},
};

export default shilpConfig;
```
