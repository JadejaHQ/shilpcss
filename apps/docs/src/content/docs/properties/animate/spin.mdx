## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		animate: {
			spin: {
				in: {
					property: "--entr-rtt: <n><v><i>;",
					themeKey: "angles",
					values: {
						DEFAULT: "0deg",
					},
				},

				out: {
					property: "--ext-rtt: <n><v><i>;",
					themeKey: "angles",
					values: {
						DEFAULT: "0deg",
					},
				},
			},
		},
	},
};

export default shilpConfig;
```
