## Reference

```js title="shilp.config.js"
const shilpConfig = {
	source: "react",

	properties: {
		live: {
			caret: {
				property: "caret-color: <v><i>;",
				resolve: "color",
				themeKey: "colors",
				variant: true,
				values: {
					auto: "auto",
				},
			},
		},
	},
};
```
