## Reference

<Alert>

Must see:
[Youtube: CSS Siblings Selector](https://www.youtube.com/watch?v=rndMS4pEKP8&t=145s)

</Alert>

```js title="shilp.config.js"
const shilpConfig = {
	target: "react",

	mixins: {
		siblings: {
			// https://www.youtube.com/watch?v=rndMS4pEKP8&t=145s

			/* functions */
			match: "& ~ <1>", // not nested, not from other parents, all following
			"match-closest": "& + <1>", // not nested, not from other parents, all direct following
		},
	},
};

export default shilpConfig;
```
